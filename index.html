Highlight Points, Lines & Polygons
// GeoJSON for points, lines, and polygons const data = { type: 'FeatureCollection', features: [ { type: 'Feature', properties: { name: 'Wildlife Crossing A', type: 'point' }, geometry: { type: 'Point', coordinates: [-79.38, 43.66] } }, { type: 'Feature', properties: { name: 'Corridor Line', type: 'line' }, geometry: { type: 'LineString', coordinates: [ [-79.4, 43.65], [-79.35, 43.67] ] } }, { type: 'Feature', properties: { name: 'Habitat Area', type: 'polygon' }, geometry: { type: 'Polygon', coordinates: [[ [-79.38, 43.64], [-79.36, 43.64], [-79.36, 43.66], [-79.38, 43.66], [-79.38, 43.64] ]] } } ] }; map.on('load', () => { map.addSource('features', { type: 'geojson', data }); // Add fill layer (polygon) map.addLayer({ id: 'polygon-fill', type: 'fill', source: 'features', paint: { 'fill-color': '#088', 'fill-opacity': 0.4 }, filter: ['==', ['get', 'type'], 'polygon'] }); // Add line layer map.addLayer({ id: 'line-layer', type: 'line', source: 'features', paint: { 'line-color': '#800080', 'line-width': 4 }, filter: ['==', ['get', 'type'], 'line'] }); // Add circle layer for points map.addLayer({ id: 'point-layer', type: 'circle', source: 'features', paint: { 'circle-radius': 8, 'circle-color': '#ff6600' }, filter: ['==', ['get', 'type'], 'point'] }); // Add highlight on hover ['polygon-fill', 'line-layer', 'point-layer'].forEach(layerId => { map.on('mouseenter', layerId, e => { map.getCanvas().style.cursor = 'pointer'; const feature = e.features[0]; const coordinates = feature.geometry.type === 'Point' ? feature.geometry.coordinates : e.lngLat.toArray(); const name = feature.properties.name; new mapboxgl.Popup() .setLngLat(coordinates) .setHTML(`${name}`) .addTo(map); }); map.on('mouseleave', layerId, () => { map.getCanvas().style.cursor = ''; const popups = document.getElementsByClassName('mapboxgl-popup'); if (popups.length) popups[0].remove(); }); }); });
